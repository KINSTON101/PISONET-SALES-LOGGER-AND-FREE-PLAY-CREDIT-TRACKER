<script>
    // MAO NI UTOK SATUNG SYSTEM JAVASCRIPT kung wala ni dili mo gana ang mga buttonds
let savedAmount = "";
let tempCustName = "";
let selectedRole = "";
let customerData = [];


function showSection(id) {
    document.querySelectorAll('.section').forEach(sec => sec.classList.remove('active'));
    document.getElementById(id).classList.add('active');
    if(id === 'trackCustomer') searchCustomer();
}

// LOGIN
function openInputs(role) {
    selectedRole = role;
    document.getElementById('roleSelect').style.display = 'none';
    document.getElementById('credentials').style.display = 'block';
}
function resetModal() {
    document.getElementById('roleSelect').style.display = 'block';
    document.getElementById('credentials').style.display = 'none';
}
function finishLogin() {
    document.getElementById('toggle-login').checked = false;
    showSection(selectedRole);
    resetModal();
}


// ADD CUSTOMER FUNCTIONS
function handleCustName(e) {
    if (e.key === 'Enter' && e.target.value.trim() !== "") {
        tempCustName = e.target.value;
        document.getElementById('spanName').innerText = tempCustName;
        document.getElementById('nameInputPart').style.display = 'none';
        document.getElementById('idInputPart').style.display = 'block';
        document.getElementById('custIdInput').focus();
    }
}

function handleCustId(e) {
    if (e.key === 'Enter' && e.target.value.trim() !== "") {
        let id = e.target.value;
        let time = new Date().toLocaleString();
        customerData.push({ name: tempCustName, id: id, status: "CLAIMED", time: time });
        alert("Saved: " + tempCustName);
        resetAddCust();
        showSection('trackCustomer');
    }
}

function resetAddCust() {
    tempCustName = "";
    document.getElementById('custNameInput').value = "";
    document.getElementById('custIdInput').value = "";
    document.getElementById('nameInputPart').style.display = 'block';
    document.getElementById('idInputPart').style.display = 'none';
}

// TRACK OG SEARCH
function searchCustomer() {
    let query = document.getElementById('searchBar').value.toLowerCase();
    let display = document.getElementById('customerListDisplay');
    let recordsDisplay = document.getElementById('allRecordsList');
    display.innerHTML = "";
    recordsDisplay.innerHTML = "";
    let filtered = customerData.filter(c => c.name.toLowerCase().includes(query) || c.id.toLowerCase().includes(query));
    if (filtered.length === 0) {
        display.innerHTML = "<p>No results found.</p>";
        recordsDisplay.innerHTML = "<p>No transactions recorded today.</p>";
    } else {
        filtered.forEach(c => {
            let html = `
                <div class="customer-card">
                    <b>Name:</b> ${c.name}<br>
                    <b>ID:</b> ${c.id}<br>
                    <b>Status:</b> <span style="color:#00f2fe;">${c.status}</span><br>
                    <small><b>Added:</b> ${c.time}</small>
                </div>`;
            display.innerHTML += html;
            recordsDisplay.innerHTML += html;
        });
    }
}

function deleteRecords() {
    if(confirm("Delete all records?")) {
        customerData = [];
        searchCustomer();
    }
}

// SALES LOGGER
function handleSalesAmount(e) {
    if (e.key === 'Enter' && e.target.value.trim() !== "") {
        savedAmount = e.target.value;
        document.getElementById('step1').style.display = 'none';
        document.getElementById('step2').style.display = 'block';
        document.getElementById('salesConfirm').focus();
    }
}
function handleSalesConfirm(e) {
    if (e.key === 'Enter') {
        let choice = e.target.value.toUpperCase();
        if (choice === 'Y') { alert("Sales Added: ₱" + savedAmount); resetSales(); showSection('admin'); }
        else if (choice === 'N') { resetSales(); }
        else { alert("Please type Y or N"); e.target.value = ""; }
    }
}
function resetSales() {
    savedAmount = "";
    document.getElementById('salesAmount').value = "";
    document.getElementById('salesConfirm').value = "";
    document.getElementById('step1').style.display = 'block';
    document.getElementById('step2').style.display = 'none';
}
